def data_gen(date1,date2,cates):
    pipelines=[
    {'$match':{'$and':[
                {'pub_date':{'$gte':date1,'$lte':date2}},
                {'cates':{'$all':[cates]}},
                {'look':{'$nin':['-']}}
            ]}},
    {'$group':{'_id':'$look','avg_price':{'$avg':'$price'}}},
    {'$sort':{'avg_price':1}}
]
    for i in item_info.aggregate(pipelines):
        yield i['avg_price']
